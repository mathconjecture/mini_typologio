 \section{Τύπος Taylor}
 \[
     f(x) = f(a) + \frac{f'(a)}{1!} (x-a) + \frac{f''(a)}{2!} (x-a)^{2} + \cdots + 
     \frac{f^{(n)}(a)}{n!} (x-a)^{n} + R_{n}(x)
  \] 
  
  \section{Τύπος Υπολοίπου κατά Lagrange}
  \[
      R_{n}(x) = \frac{f^{(n+1)}(\xi)}{(n+1)!} (x-a)^{n+1},
      \quad \text{όπου} \; \xi \in (a,x)
      \; \text{ή} \; \xi \in (x,a)
   \] 

  \section{Τύπος Maclaurin}
  \[
      f(x) = f(0) + \frac{f'(0)}{1!} x + \frac{f''(0)}{2!} x^{2} + \cdots + 
      \frac{f^{(n)}(0)}{n!} x^{n} + R_{n}(x)
   \]
