\pgfplotsset{
  myaxis/.style={axis lines=center,axis line style={thick,blue!50,-stealth},tick label
  style={font=\small,blue!50},xtick=\empty,ytick=\empty,tick style={blue!50}},
myplot/.style={Col1!75,ultra thick,samples=500,no marks,smooth,},
  dashed lines/.style={dashed,blue!50,ultra thin},
}
\tikzset{dashed lines/.style={dashed,blue!50,ultra thin}}


\pgfmathdeclarefunction{asinh}{1}{%
  \pgfmathparse{ln(#1 + sqrt((#1)^2 + 1))}%
}
\pgfmathdeclarefunction{acosh}{1}{%
  \pgfmathparse{ln(#1 + sqrt((#1)^2 - 1))}%
}

\pgfkeys{/pgf/declare function={atanh(\x) = 0.5*(ln((1+\x)/(1-\x)));}}
\pgfkeys{/pgf/declare function={acoth(\x) = 0.5*(ln((\x+1)/(\x-1)));}}
